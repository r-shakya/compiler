
integer factorial(integer  n){
	integer one;
	one = 1;
	if(n <= 0){
		send one;
	}
	integer factn;
	n = n - 1;
	factn = (n + 1)*factn;
	send factn; 
}

void main()
{
  integer n;
  output("Enter Val :");
  input( n );
  output("Val of Fact(");
  output(n);
  output(") : ");
  integer factn;
  factn = call("factorial",n);
  output(factn);
}
