

void main()
{
  integer array[7], n, c, d, swap;
  n = 7;
  for (c = 0; c < n; c = c + 1){
     array[c] = 6 - c;
  }
  
  output("hello world");
  input(n);

  for (c = 0 ; c < n - 1; c = c + 1)
  {
    for (d = 0 ; d < (n - c) - 1; d = d + 1)
    {
      if (array[d] > array[d+1])
      {
        swap       = array[d];
        array[d]   = array[d+1];
        array[d+1] = swap;
      }
    }
  }

}
