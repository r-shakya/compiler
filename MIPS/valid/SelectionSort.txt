void main(){
    integer a[5],temp,index,i,j,n;
    n=5;
    for (i = 0; i < n; i = i + 1){
       input( a[i] );
    }
    
    for(i=0; i<n-1; i=i+1){
        index=i;
        for(j=i+1; j<n; j=j+1){
            if(a[index]>a[j]){
                index=j;
            }
        }
        temp=a[i];
        a[i]=a[index];
        a[index]=temp;
    }
    
    for (i = 0; i < n; i = i + 1){
       output( a[i] );
       output("   ");
    }
}
